Ажилтаны_Мэдээллийг_Дамжуулах_процедур (Өгөгдөл_унших_JSON_файл, Өгөгдөл_Хадгалах_JSON_файл, Шилжүүлэх_ажилчдын_ID ){
    Өгөгдөл унших замаа тодорхойлно
    Өгөгдөл шилжүүлэх замаа тодорхойлно

    Өгөгдлөө_унших_функц ("Өгөгдөл унших зам") {
        IF (Алдаа гарвал) алдааг хэвлэнэ
        TRY {
            JSON дээрх өгөгдлөө PARSE хийж "sourceWorkers" хувьсагчид хадгална
            Сонгосон ажилчдыг хадгалах хувьсагч үүсгэнэ
            FOR EACH "worker" IN "sourceWorkers"{
                IF ("worker.id" IN "Шилжүүлэх_ажилчдын_ID")
                "worker" ийг хадгална
                "worker" ийг "sourceWorkers" устгана
            }
        
            Өгөгдлөө_уншина_функц("Өгөгдөл шилжүүлэх зам") {
                IF(Алдаа гарвал) алдааг хэвлэнэ
                TRY {
                    JSON дээрх өгөгдлөө PARSE хийж "destinationWorkers" хадгална
                    Сонгосон ажилчдыг "destinationWorkers" дээр нэмэж бичнэ
                    Өгөгдөл_бичих_функц("destinationWorkers"){
                        шинэчлэгдсэн "destinationWorkers" JSON файлруугаа бичнэ
                    }
                    Өгөгдөл_бичих_функц("sourceWorkers" ){
                        шинэчлэгдсэн "sourceWorkers" JSON файлруугаа бичнэ
                    }
                }
                CATCH("error"){
                    алдааг хэвлэх
                }
            }
        }
        CATCH("error"){
            алдааг хэвлэх
        }
    }
}
